#include "MyCharacter.h"
#include "Components/SkeletalMeshComponent.h"
#include "GameFramework/CharacterMovementComponent.h" // Controla el movimiento del personaje.
#include "Components/CapsuleComponent.h" // Componente de colisi√≥n del personaje.
#include "Animation/AnimInstance.h" // Para asignar Animation Blueprints.
#include "UObject/ConstructorHelpers.h" // Permite cargar recursos desde el editor.
#include "Core/I_Interactuable.h" // Interfaz para objetos interactuables.

// Constructor
AMyCharacter::AMyCharacter()
{
	// Habilita Tick() cada frame (se puede desactivar para optimizar).
	PrimaryActorTick.bCanEverTick = true;

	// üìå Configuraci√≥n del Skeletal Mesh
	static ConstructorHelpers::FObjectFinder DefaultMesh(TEXT("/Game/Characters/Mannequins/Meshes/SKM_Manny_Simple.SKM_Manny_Simple"));
	if (DefaultMesh.Succeeded())
	{
		UE_LOG(LogTemp, Warning, TEXT("‚úÖ Default SkeletalMesh Loaded Successfully!"));
		GetMesh()->SetSkeletalMesh(DefaultMesh.Object);
		GetMesh()->SetRelativeLocation(FVector(0.0f, 0.0f, -90.0f));
		GetMesh()->SetRelativeRotation(FRotator(0.0f, -90.0f, 0.0f));
	}
	else
	{
		UE_LOG(LogTemp, Error, TEXT("‚ùå Failed to Load Default SkeletalMesh!"));
	}

	// üìå Configuraci√≥n del Animation Blueprint
	static ConstructorHelpers::FClassFinder DefaultAnimBP(TEXT("/Game/Characters/Mannequins/Animations/ABP_Character.ABP_Character_C"));
	if (DefaultAnimBP.Succeeded())
	{
		UE_LOG(LogTemp, Warning, TEXT("‚úÖ Default Animation Blueprint Loaded Successfully!"));
		GetMesh()->SetAnimationMode(EAnimationMode::AnimationBlueprint);
		GetMesh()->SetAnimInstanceClass(DefaultAnimBP.Class);
	}
	else
	{
		UE_LOG(LogTemp, Error, TEXT("‚ùå Failed to Load Animation Blueprint!"));
	}

	// üìå Configuraci√≥n del movimiento del personaje
	GetCharacterMovement()->bOrientRotationToMovement = true; // El personaje rota hacia la direcci√≥n del movimiento.
	bUseControllerRotationYaw = false;
	bUseControllerRotationPitch = false;
	bUseControllerRotationRoll = false;

	// üìå Configuraci√≥n de la c√°mara
	SpringArmComp = CreateDefaultSubobject(TEXT("SpringArmComp"));
	SpringArmComp->SetupAttachment(RootComponent);
	SpringArmComp->TargetArmLength = 300.0f; // Distancia de la c√°mara al personaje.
	SpringArmComp->bUsePawnControlRotation = true; // Usa la rotaci√≥n del control del jugador.

	CameraComp = CreateDefaultSubobject(TEXT("CameraComp"));
	CameraComp->SetupAttachment(SpringArmComp, USpringArmComponent::SocketName);

	// üìå Configuraci√≥n del CapsuleComponent (ya existente en ACharacter)
	UCapsuleComponent* Capsule = GetCapsuleComponent();
	if (Capsule)
	{
		Capsule->InitCapsuleSize(42.0f, 96.0f);
		Capsule->SetCapsuleHalfHeight(100.f);

		// Configuraci√≥n de colisi√≥n
		Capsule->SetCollisionEnabled(ECollisionEnabled::QueryAndPhysics);
		Capsule->SetCollisionObjectType(ECollisionChannel::ECC_Pawn);
		Capsule->SetCollisionResponseToAllChannels(ECollisionResponse::ECR_Block);

		UE_LOG(LogTemp, Log, TEXT("‚úÖ CapsuleComponent Configured"));
	}
}

// üìå Llamado cuando el personaje es generado
void AMyCharacter::BeginPlay()
{
	Super::BeginPlay();
}

// üìå Llamado cada frame
void AMyCharacter::Tick(float DeltaTime)
{
	Super::Tick(DeltaTime);
}

// üìå Configuraci√≥n de controles del personaje
void AMyCharacter::SetupPlayerInputComponent(UInputComponent* PlayerInputComponent)
{
	if (!PlayerInputComponent) return;

	Super::SetupPlayerInputComponent(PlayerInputComponent);

	PlayerInputComponent->BindAxis("MoveForward", this, &AMyCharacter::MoveForward);
	PlayerInputComponent->BindAxis("MoveRight", this, &AMyCharacter::MoveRight);
	PlayerInputComponent->BindAxis("LookUp", this, &AMyCharacter::LookUp);
	PlayerInputComponent->BindAxis("Turn", this, &AMyCharacter::Turn);
}

// üìå Movimiento hacia adelante/atr√°s
void AMyCharacter::MoveForward(float Value)
{
	if (Controller && Value != 0.0f)
	{
		// Obtiene la rotaci√≥n actual del controlador y solo usa la direcci√≥n hacia adelante
		const FRotator Rotation = Controller->GetControlRotation();
		const FVector Direction = FRotationMatrix(Rotation).GetUnitAxis(EAxis::X);
		AddMovementInput(Direction, Value);
	}
}

// üìå Movimiento hacia la derecha/izquierda
void AMyCharacter::MoveRight(float Value)
{
	if (Controller && Value != 0.0f)
	{
		// Obtiene la rotaci√≥n actual del controlador y solo usa la direcci√≥n hacia la derecha
		const FRotator Rotation = Controller->GetControlRotation();
		const FVector Direction = FRotationMatrix(Rotation).GetUnitAxis(EAxis::Y);
		AddMovementInput(Direction, Value);
	}
}

// üìå Movimiento de la c√°mara en vertical
void AMyCharacter::LookUp(float Value)
{
	AddControllerPitchInput(Value);
}

// üìå Movimiento de la c√°mara en horizontal
void AMyCharacter::Turn(float Value)
{
	AddControllerYawInput(Value);
}


// üìå Funci√≥n para interactuar con objetos en el mundo
void AMyCharacter::Interactuar()
{
    if (CurrentInteractableActor && CurrentInteractableActor->GetClass()->ImplementsInterface(UI_Interactuable::StaticClass()))
    {
        UE_LOG(LogTemp, Log, TEXT("Interactuando con: %s"), *CurrentInteractableActor->GetName());
        II_Interactuable::Execute_Interact(CurrentInteractableActor, this);
    }
    else
    {
        UE_LOG(LogTemp, Warning, TEXT("No hay ning√∫n actor interactuable o no implementa la interfaz."));
    }
}
